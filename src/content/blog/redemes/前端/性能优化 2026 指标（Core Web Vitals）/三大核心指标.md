## 一、Largest Contentful Paint (LCP)

### 1. LCP 优化策略
```javascript
// LCP 监控与优化器
class LCPOptimizer {
    constructor() {
        this.lcpElement = null;
        this.lcpTime = 0;
        this.observer = null;
        this.optimizations = new Map();
    }

    // 开始监控 LCP
    startMonitoring() {
        // 使用 PerformanceObserver 监控 LCP
        this.observer = new PerformanceObserver((list) => {
            const entries = list.getEntries();
            const lcpEntry = entries[entries.length - 1];
            
            this.lcpTime = lcpEntry.startTime;
            this.lcpElement = this.getElementFromEntry(lcpEntry);
            
            this.analyzeLCP(lcpEntry);
            this.triggerOptimizations();
        });

        this.observer.observe({ type: 'largest-contentful-paint', buffered: true });
        
        // 添加备用监控
        this.addFallbackMonitoring();
    }

    // 分析 LCP 性能
    analyzeLCP(lcpEntry) {
        const analysis = {
            elementType: lcpEntry.element?.tagName || 'unknown',
            resourceType: this.getResourceType(lcpEntry),
            loadTime: lcpEntry.startTime,
            size: lcpEntry.size,
            url: lcpEntry.url || 'inline'
        };

        // 检查是否满足性能标准
        analysis.meetsStandard = lcpEntry.startTime <= 2500; // 2.5秒标准
        analysis.grade = this.calculateGrade(lcpEntry.startTime);
        
        console.log('LCP 分析:', analysis);
        return analysis;
    }

    // 计算 LCP 等级
    calculateGrade(lcpTime) {
        if (lcpTime <= 2500) return 'good';
        if (lcpTime <= 4000) return 'needs-improvement';
        return 'poor';
    }

    // 获取资源类型
    getResourceType(lcpEntry) {
        if (lcpEntry.url) {
            if (lcpEntry.url.endsWith('.jpg') || lcpEntry.url.endsWith('.png')) {
                return 'image';
            }
            if (lcpEntry.url.endsWith('.css')) {
                return 'stylesheet';
            }
            if (lcpEntry.url.endsWith('.js')) {
                return 'script';
            }
        }
        return 'text';
    }

    // 触发优化策略
    triggerOptimizations() {
        const optimizations = [
            this.optimizeLCPImage.bind(this),
            this.optimizeLCPText.bind(this),
            this.optimizeResourceLoading.bind(this)
        ];

        for (const optimization of optimizations) {
            optimization();
        }
    }

    // 优化 LCP 图片
    optimizeLCPImage() {
        if (!this.lcpElement || this.lcpElement.tagName !== 'IMG') return;

        const img = this.lcpElement;
        
        // 检查图片是否已优化
        if (img.hasAttribute('data-optimized')) return;

        // 应用优化策略
        this.applyImageOptimizations(img);
        img.setAttribute('data-optimized', 'true');
    }

    // 应用图片优化
    applyImageOptimizations(img) {
        // 设置加载优先级
        img.loading = 'eager';
        img.fetchPriority = 'high';
        
        // 预加载关键图片
        if (!img.complete) {
            const link = document.createElement('link');
            link.rel = 'preload';
            link.as = 'image';
            link.href = img.src;
            document.head.appendChild(link);
        }

        // 添加响应式图片支持
        this.addResponsiveImages(img);
    }

    // 添加响应式图片
    addResponsiveImages(img) {
        const srcset = img.getAttribute('srcset');
        if (!srcset && img.src) {
            // 根据图片尺寸生成响应式版本
            const sizes = this.generateResponsiveSizes(img);
            img.srcset = sizes.map(size => `${size.url} ${size.width}w`).join(', ');
            img.sizes = '(max-width: 768px) 100vw, 50vw';
        }
    }

    // 生成响应式尺寸
    generateResponsiveSizes(img) {
        const baseUrl = img.src;
        const sizes = [320, 640, 1024, 1440, 1920];
        
        return sizes.map(width => ({
            url: this.generateResponsiveUrl(baseUrl, width),
            width
        }));
    }

    generateResponsiveUrl(baseUrl, width) {
        // 实际实现中会根据图片服务生成响应式 URL
        return `${baseUrl}?width=${width}`;
    }

    // 优化 LCP 文本内容
    optimizeLCPText() {
        if (!this.lcpElement || this.lcpElement.tagName !== 'DIV') return;

        const element = this.lcpElement;
        
        // 检查字体加载
        this.optimizeFontLoading(element);
        
        // 优化文本渲染
        this.optimizeTextRendering(element);
    }

    // 优化字体加载
    optimizeFontLoading(element) {
        const fontFamily = getComputedStyle(element).fontFamily;
        
        // 预加载关键字体
        if (fontFamily && !fontFamily.includes('system-ui')) {
            const link = document.createElement('link');
            link.rel = 'preload';
            link.as = 'font';
            link.href = this.getFontUrl(fontFamily);
            link.crossOrigin = 'anonymous';
            document.head.appendChild(link);
        }
    }

    // 获取字体 URL
    getFontUrl(fontFamily) {
        // 根据字体家族生成字体文件 URL
        const fontMap = {
            'Roboto': '/fonts/roboto.woff2',
            'Open Sans': '/fonts/open-sans.woff2'
        };
        
        for (const [font, url] of Object.entries(fontMap)) {
            if (fontFamily.includes(font)) {
                return url;
            }
        }
        
        return null;
    }

    // 添加备用监控
    addFallbackMonitoring() {
        // 如果 PerformanceObserver 不可用，使用备用方案
        if (!window.PerformanceObserver) {
            this.fallbackLCPMonitoring();
        }
    }

    fallbackLCPMonitoring() {
        let lcpCandidate = null;
        
        const checkLCP = () => {
            const elements = document.querySelectorAll('img, video, div[class*="hero"], h1, h2');
            
            for (const element of elements) {
                const rect = element.getBoundingClientRect();
                const area = rect.width * rect.height;
                
                if (!lcpCandidate || area > lcpCandidate.area) {
                    lcpCandidate = { element, area, time: performance.now() };
                }
            }
            
            this.lcpTime = lcpCandidate?.time || 0;
            this.lcpElement = lcpCandidate?.element || null;
        };

        // 定期检查
        setInterval(checkLCP, 100);
        
        // 页面加载完成后最终检查
        window.addEventListener('load', () => {
            setTimeout(checkLCP, 1000);
        });
    }
}
```

```javascript
class AdvancedLCPOptimization {

            estimatedImprovement: 0,
            recommendations: []
        };

        for (const result of results) {
            if (result.success) {
                analysis.estimatedImprovement += result.estimatedImprovement || 0;
                analysis.recommendations.push(...(result.recommendations || []));
            }
        }

        return analysis;
    }

    // 实时 LCP 预测
    predictLCP() {
        const currentTime = performance.now();
        const resources = performance.getEntriesByType('resource');
        
        const criticalResources = resources.filter(resource => 
            resource.name.includes('.css') || 
            resource.name.includes('.js') ||
            resource.name.match(/\.(jpg|png|webp)$/)
        );

        // 预测 LCP 时间
        let predictedLCP = currentTime;
        
        for (const resource of criticalResources) {
            if (!resource.responseEnd) {
                // 资源尚未加载完成，预测其完成时间
                predictedLCP = Math.max(predictedLCP, currentTime + this.estimateRemainingTime(resource));
            }
        }

        return predictedLCP;
    }

    estimateRemainingTime(resource) {
        // 基于已传输字节和网络速度估算剩余时间
        const transferred = resource.transferSize || 0;
        const totalSize = resource.encodedBodySize || resource.decodedBodySize || 0;
        
        if (totalSize === 0) return 0;
        
        const progress = transferred / totalSize;
        const elapsed = performance.now() - resource.startTime;
        const estimatedTotal = elapsed / progress;
        
        return estimatedTotal - elapsed;
    }
}
```

### 1. FID 监控与优化
```javascript
// FID 优化管理器
class FIDOptimizer {
    constructor() {
        this.fidEntries = [];
        this.longTasks = [];
        this.observer = null;
        this.optimizationQueue = [];
    }

    // 开始监控 FID
    startMonitoring() {
        // 监控 FID
        this.observer = new PerformanceObserver((list) => {
            const entries = list.getEntries();
            
            for (const entry of entries) {
                if (entry.entryType === 'first-input') {
                    this.handleFID(entry);
                } else if (entry.entryType === 'longtask') {
                    this.handleLongTask(entry);
                }
            }
        });

        this.observer.observe({
            type: ['first-input', 'longtask'],
            buffered: true
        });

        // 监控主线程阻塞
        this.monitorMainThread();
    }

    // 处理 FID 事件
    handleFID(entry) {
        this.fidEntries.push(entry);
        
        const analysis = {
            delay: entry.processingStart - entry.startTime,
            target: entry.target?.tagName || 'unknown',
            eventType: entry.name,
            timestamp: entry.startTime
        };

        analysis.grade = this.calculateFIDGrade(analysis.delay);
        
        console.log('FID 事件:', analysis);
        
        // 如果 FID 过长，触发优化
        if (analysis.delay > 100) {
            this.scheduleOptimization(analysis);
        }
    }

    // 计算 FID 等级
    calculateFIDGrade(delay) {
        if (delay <= 100) return 'good';
        if (delay <= 300) return 'needs-improvement';
        return 'poor';
    }

    // 处理长任务
    handleLongTask(entry) {
        this.longTasks.push(entry);
        
        // 分析长任务对 FID 的影响
        this.analyzeLongTaskImpact(entry);
    }

    // 分析长任务影响
    analyzeLongTaskImpact(longTask) {
        const overlappingFID = this.fidEntries.find(fid => 
            fid.startTime >= longTask.startTime && 
            fid.startTime <= longTask.startTime + longTask.duration
        );

        if (overlappingFID) {
            console.warn('长任务导致 FID 延迟:', {
                longTaskDuration: longTask.duration,
                fidDelay: overlappingFID.processingStart - overlappingFID.startTime,
                overlap: overlappingFID.startTime - longTask.startTime
            });
        }
    }

    // 调度优化
    scheduleOptimization(analysis) {
        const optimization = {
            type: 'fid_optimization',
            analysis,
            priority: analysis.delay > 300 ? 'high' : 'medium',
            scheduledAt: Date.now()
        };

        this.optimizationQueue.push(optimization);
        this.processOptimizationQueue();
    }

    // 处理优化队列
    processOptimizationQueue() {
        if (this.optimizationQueue.length === 0) return;

        // 按优先级排序
        this.optimizationQueue.sort((a, b) => {
            const priorityOrder = { high: 0, medium: 1, low: 2 };
            return priorityOrder[a.priority] - priorityOrder[b.priority];
        });

        // 执行最高优先级的优化
        const optimization = this.optimizationQueue.shift();
        this.executeOptimization(optimization);
    }

    // 执行优化
    executeOptimization(optimization) {
        const strategies = [
            this.optimizeJavaScriptExecution.bind(this),
            this.deferNonCriticalWork.bind(this),
            this.optimizeEventHandlers.bind(this)
        ];

        for (const strategy of strategies) {
            strategy(optimization.analysis);
        }

        console.log('FID 优化执行完成:', optimization);
    }

    // 优化 JavaScript 执行
    optimizeJavaScriptExecution(analysis) {
        // 拆分长任务
        this.splitLongTasks();
        
        // 优化事件循环
        this.optimizeEventLoop();
    }

    // 拆分长任务
    splitLongTasks() {
        // 使用 setTimeout 或 requestIdleCallback 拆分任务
        if (typeof requestIdleCallback === 'function') {
            requestIdleCallback(() => {
                this.executeDeferredWork();
            });
        } else {
            setTimeout(() => {
                this.executeDeferredWork();
            }, 0);
        }
    }

    // 监控主线程
    monitorMainThread() {
        let lastTime = performance.now();
        
        const checkBlocking = () => {
            const currentTime = performance.now();
            const delta = currentTime - lastTime;
            
            // 如果主线程被阻塞超过 50ms
            if (delta > 50) {
                console.warn('主线程阻塞 detected:', delta + 'ms');
                this.triggerProactiveOptimization();
            }
            
            lastTime = currentTime;
            setTimeout(checkBlocking, 10);
        };
        
        checkBlocking();
    }

    // 触发主动优化
    triggerProactiveOptimization() {
        // 在用户交互前主动优化
        this.preemptiveOptimization();
    }
}
```

```javascript
// 任务调度器
class TaskScheduler {
    constructor() {
        this.priorityQueue = [];
        this.currentTask = null;
    }

    setupPriorityQueue() {
        // 重写 setTimeout 和 setInterval 以支持优先级
        this.overrideTimers();
        
        // 监控 Promise 执行
        this.monitorPromises();
    }

    overrideTimers() {
        const originalSetTimeout = window.setTimeout;
        const originalSetInterval = window.setInterval;

        window.setTimeout = (callback, delay, ...args) => {
            return this.scheduleTask(callback, delay, 'timeout', args);
        };

        window.setInterval = (callback, delay, ...args) => {
            return this.scheduleTask(callback, delay, 'interval', args);
        };
    }

    scheduleTask(callback, delay, type, args) {
        const task = {
            callback,
            delay,
            type,
            args,
            priority: this.calculatePriority(callback, delay),
            scheduledAt: Date.now()
        };

        this.priorityQueue.push(task);
        this.priorityQueue.sort((a, b) => a.priority - b.priority);
        
        return this.executeTask(task);
    }

    calculatePriority(callback, delay) {
        let priority = 5; // 默认优先级
        
        // 根据延迟时间调整优先级
        if (delay === 0) priority = 1;
        else if (delay < 16) priority = 2; // 一帧内
        else if (delay < 100) priority = 3;
        
        // 根据回调函数内容调整优先级
        const callbackStr = callback.toString();
        if (callbackStr.includes('UI') || callbackStr.includes('render')) {
            priority = 1; // UI 相关任务最高优先级
        }
        
        return priority;
    }

    executeTask(task) {
        if (task.type === 'timeout') {
            return setTimeout(() => {
                this.runTask(task);
            }, task.delay);
        } else {
            return setInterval(() => {
                this.runTask(task);
            }, task.delay);
        }
    }

    runTask(task) {
        const startTime = performance.now();
        
        try {
            task.callback(...task.args);
        } catch (error) {
            console.error('任务执行错误:', error);
        }
        
        const duration = performance.now() - startTime;
        
        // 记录任务执行时间
        this.recordTaskExecution(task, duration);
    }

    recordTaskExecution(task, duration) {
        // 记录任务执行统计
        console.log('任务执行记录:', {
            type: task.type,
            priority: task.priority,
            duration,
            timestamp: Date.now()
        });
    }
}
```

## 三、Cumulative Layout Shift (CLS)

### 1. CLS 监控与优化
```javascript
// CLS 优化管理器
class CLSOptimizer {
    constructor() {
        this.clsScore = 0;
        this.layoutShifts = [];
        this.observer = null;
        this.elementTracker = new Map();
    }

    // 开始监控 CLS
    startMonitoring() {
        this.observer = new PerformanceObserver((list) => {
            for (const entry of list.getEntries()) {
                if (entry.hadRecentInput) continue;
                
                this.clsScore += entry.value;
                this.layoutShifts.push(entry);
                
                this.analyzeLayoutShift(entry);
                this.triggerCLSOptimization(entry);
            }
        });

        this.observer.observe({ type: 'layout-shift', buffered: true });
        
        // 监控可能导致布局变化的元素
        this.monitorLayoutTriggers();
    }

    // 分析布局偏移
    analyzeLayoutShift(entry) {
        const analysis = {
            score: entry.value,
            cumulativeScore: this.clsScore,
            sources: entry.sources || [],
            timestamp: entry.startTime
        };

        analysis.grade = this.calculateCLSGrade(this.clsScore);
        analysis.impact = this.assessImpact(entry);
        
        console.log('布局偏移分析:', analysis);
        return analysis;
    }

    // 计算 CLS 等级
    calculateCLSGrade(score) {
        if (score <= 0.1) return 'good';
        if (score <= 0.25) return 'needs-improvement';
        return 'poor';
    }

    // 评估影响
    assessImpact(entry) {
        let impact = 'low';
        
        if (entry.sources) {
            for (const source of entry.sources) {
                if (source.node && this.isVisibleElement(source.node)) {
                    const rect = source.node.getBoundingClientRect();
                    const viewportArea = window.innerWidth * window.innerHeight;
                    const elementArea = rect.width * rect.height;
                    
                    if (elementArea > viewportArea * 0.25) {
                        impact = 'high';
                        break;
                    } else if (elementArea > viewportArea * 0.1) {
                        impact = 'medium';
                    }
                }
            }
        }
        
        return impact;
    }

    // 触发 CLS 优化
    triggerCLSOptimization(entry) {
        if (entry.value > 0.05) { // 显著的布局偏移
            this.applyCLSOptimizations(entry);
        }
    }

    // 应用 CLS 优化
    applyCLSOptimizations(entry) {
        const optimizations = [
            this.optimizeImages.bind(this),
            this.optimizeAds.bind(this),
            this.optimizeEmbeds.bind(this),
            this.optimizeDynamicContent.bind(this)
        ];

        for (const optimization of optimizations) {
            optimization(entry);
        }
    }

    // 优化图片导致的布局偏移
    optimizeImages(entry) {
        const imageSources = entry.sources?.filter(source => 
            source.node?.tagName === 'IMG'
        ) || [];

        for (const source of imageSources) {
            const img = source.node;
            
            // 确保图片有尺寸
            if (!img.width || !img.height) {
                this.setImageDimensions(img);
            }
            
            // 使用 aspect-ratio
            this.applyAspectRatio(img);
        }
    }

    // 设置图片尺寸
    setImageDimensions(img) {
        if (img.naturalWidth && img.naturalHeight) {
            img.width = img.naturalWidth;
            img.height = img.naturalHeight;
        } else {
            // 使用默认占位尺寸
            img.width = img.width || 300;
            img.height = img.height || 200;
        }
    }

    // 应用宽高比
    applyAspectRatio(img) {
        if (img.width && img.height) {
            const aspectRatio = img.width / img.height;
            img.style.aspectRatio = `${img.width} / ${img.height}`;
        }
    }

    // 监控布局触发器
    monitorLayoutTriggers() {
        // 监控可能导致布局变化的操作
        this.monitorDOMChanges();
        this.monitorStyleChanges();
        this.monitorFontLoading();
    }

    // 监控 DOM 变化
    monitorDOMChanges() {
        const observer = new MutationObserver((mutations) => {
            for (const mutation of mutations) {
                if (mutation.type === 'childList') {
                    this.analyzeDOMChange(mutation);
                }
            }
        });

        observer.observe(document.body, {
            childList: true,
            subtree: true
        });
    }

    // 分析 DOM 变化
    analyzeDOMChange(mutation) {
        for (const node of mutation.addedNodes) {
            if (node.nodeType === Node.ELEMENT_NODE) {
                this.assessLayoutImpact(node);
            }
        }
    }

    // 评估布局影响
    assessLayoutImpact(element) {
        // 检查元素是否可能引起布局变化
        if (this.isLayoutSensitive(element)) {
            this.applyPreventiveMeasures(element);
        }
    }

    // 检查是否布局敏感元素
    isLayoutSensitive(element) {
        const sensitiveTags = ['IMG', 'IFRAME', 'VIDEO', 'OBJECT', 'EMBED'];
        const sensitiveClasses = ['ad', 'banner', 'widget', 'embed'];
        
        if (sensitiveTags.includes(element.tagName)) {
            return true;
        }
        
        for (const cls of sensitiveClasses) {
            if (element.className.includes(cls)) {
                return true;
            }
        }
        
        return false;
    }

    // 应用预防措施
    applyPreventiveMeasures(element) {
        // 预留空间
        this.reserveSpace(element);
        
        // 添加加载状态
        this.addLoadingState(element);
    }

    // 预留空间
    reserveSpace(element) {
        const rect = element.getBoundingClientRect();
        if (rect.width === 0 || rect.height === 0) {
            // 使用数据属性设置预期尺寸
            const expectedWidth = element.getAttribute('data-width') || '300';
            const expectedHeight = element.getAttribute('data-height') || '200';
            
            element.style.width = expectedWidth + 'px';
            element.style.height = expectedHeight + 'px';
            element.style.visibility = 'hidden';
            
            // 加载完成后显示
            element.addEventListener('load', () => {
                element.style.visibility = 'visible';
            });
        }
    }
}
```

三大核心指标的优化需要综合考虑页面加载、交互响应和视觉稳定性，通过系统化的监控和优化策略，可以显著提升用户体验。
