## 一、基础概念

### 1. 断点续传原理
```javascript
// 断点续传的核心思想
class FileUploader {
    constructor(file, chunkSize = 1024 * 1024) { // 默认1MB
        this.file = file;
        this.chunkSize = chunkSize;
        this.totalChunks = Math.ceil(file.size / chunkSize);
        this.uploadedChunks = new Set();
        this.currentChunk = 0;
    }
    
    // 获取文件指纹（用于识别文件）
    async getFileHash() {
        const buffer = await this.file.slice(0, 65536).arrayBuffer();
        const hashBuffer = await crypto.subtle.digest('SHA-256', buffer);
        const hashArray = Array.from(new Uint8Array(hashBuffer));
        return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
    }
    
    // 检查上传进度
    async checkUploadProgress(fileHash) {
        const response = await fetch(`/api/upload/progress?hash=${fileHash}`);
        const progress = await response.json();
        this.uploadedChunks = new Set(progress.uploadedChunks || []);
        return progress;
    }
}
```

```javascript
// 文件分片实现
function sliceFile(file, chunkSize) {
    const chunks = [];
    let start = 0;
    
    while (start < file.size) {
        const end = Math.min(start + chunkSize, file.size);
        const chunk = file.slice(start, end);
        chunks.push({
            chunk,
            start,
            end,
            index: chunks.length,
            total: Math.ceil(file.size / chunkSize)
        });
        start = end;
    }
    
    return chunks;
}

// 分片上传
async function uploadChunk(chunkData, fileHash, chunkIndex) {
    const formData = new FormData();
    formData.append('file', chunkData.chunk);
    formData.append('hash', fileHash);
    formData.append('chunkIndex', chunkIndex);
    formData.append('totalChunks', chunkData.total);
    
    const response = await fetch('/api/upload/chunk', {
        method: 'POST',
        body: formData
    });
    
    if (!response.ok) {
        throw new Error(`上传分片 ${chunkIndex} 失败`);
    }
    
    return response.json();
}
```

## 二、核心实现

### 1. 完整的断点续传类
```javascript
class ResumableUploader {
    constructor(file, options = {}) {
        this.file = file;
        this.chunkSize = options.chunkSize || 1024 * 1024; // 1MB
        this.concurrentUploads = options.concurrentUploads || 3;
        this.retryAttempts = options.retryAttempts || 3;
        
        this.chunks = [];
        this.uploadedChunks = new Set();
        this.uploadingChunks = new Set();
        this.failedChunks = new Map(); // chunkIndex -> 失败次数
        
        this.isPaused = false;
        this.isUploading = false;
        this.fileHash = null;
        this.uploadId = null;
        
        this.onProgress = options.onProgress || (() => {});
        this.onComplete = options.onComplete || (() => {});
        this.onError = options.onError || (() => {});
    }
    
    // 初始化上传
    async init() {
        try {
            // 计算文件哈希
            this.fileHash = await this.calculateFileHash();
            
            // 检查上传状态
            const progress = await this.checkProgress();
            this.uploadedChunks = new Set(progress.uploadedChunks || []);
            this.uploadId = progress.uploadId;
            
            // 分片文件
            this.chunks = this.sliceFile();
            
            return {
                fileHash: this.fileHash,
                uploadId: this.uploadId,
                totalChunks: this.chunks.length,
                uploadedChunks: this.uploadedChunks.size
            };
        } catch (error) {
            this.onError(error);
            throw error;
        }
    }
    
    // 开始上传
    async start() {
        if (this.isUploading) return;
        
        this.isUploading = true;
        this.isPaused = false;
        
        try {
            await this.uploadChunks();
        } catch (error) {
            this.onError(error);
        }
    }
    
    // 暂停上传
    pause() {
        this.isPaused = true;
    }
    
    // 继续上传
    resume() {
        if (this.isPaused) {
            this.isPaused = false;
            this.start();
        }
    }
    
    // 取消上传
    cancel() {
        this.isUploading = false;
        this.isPaused = true;
        this.uploadingChunks.clear();
    }
    
    // 计算文件哈希
    async calculateFileHash() {
        const chunkSize = 64 * 1024; // 64KB
        const chunks = Math.ceil(this.file.size / chunkSize);
        const hash = new Uint8Array(32); // SHA-256 结果
        
        for (let i = 0; i < chunks; i++) {
            const start = i * chunkSize;
            const end = Math.min(start + chunkSize, this.file.size);
            const chunk = this.file.slice(start, end);
            const buffer = await chunk.arrayBuffer();
            
            const chunkHash = await crypto.subtle.digest('SHA-256', buffer);
            // 简单合并哈希（实际应用中需要更复杂的合并算法）
            const chunkHashArray = new Uint8Array(chunkHash);
            for (let j = 0; j < 32; j++) {
                hash[j] ^= chunkHashArray[j];
            }
        }
        
        return Array.from(hash).map(b => b.toString(16).padStart(2, '0')).join('');
    }
    
    // 分片文件
    sliceFile() {
        const chunks = [];
        let start = 0;
        
        while (start < this.file.size) {
            const end = Math.min(start + this.chunkSize, this.file.size);
            const chunk = this.file.slice(start, end);
            
            chunks.push({
                chunk,
                start,
                end,
                index: chunks.length,
                size: end - start
            });
            
            start = end;
        }
        
        return chunks;
    }
    
    // 检查上传进度
    async checkProgress() {
        const response = await fetch(`/api/upload/progress?hash=${this.fileHash}`);
        
        if (response.status === 404) {
            // 新文件，初始化上传
            const initResponse = await fetch('/api/upload/init', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    fileName: this.file.name,
                    fileSize: this.file.size,
                    fileHash: this.fileHash,
                    chunkSize: this.chunkSize
                })
            });
            
            return await initResponse.json();
        }
        
        return await response.json();
    }
    
    // 上传分片
    async uploadChunks() {
        const pendingChunks = this.chunks
            .filter(chunk => 
                !this.uploadedChunks.has(chunk.index) && 
                !this.uploadingChunks.has(chunk.index) &&
                !this.failedChunks.has(chunk.index)
            );
        
        if (pendingChunks.length === 0) {
            // 所有分片上传完成，合并文件
            await this.mergeFile();
            this.isUploading = false;
            this.onComplete();
            return;
        }
        
        // 并发上传
        const concurrentChunks = pendingChunks.slice(0, this.concurrentUploads);
        
        await Promise.allSettled(
            concurrentChunks.map(chunk => this.uploadSingleChunk(chunk))
        );
        
        // 继续上传剩余分片
        if (!this.isPaused && this.isUploading) {
            setTimeout(() => this.uploadChunks(), 0);
        }
    }
    
    // 上传单个分片
    async uploadSingleChunk(chunk) {
        if (this.isPaused || !this.isUploading) return;
        
        this.uploadingChunks.add(chunk.index);
        
        try {
            const formData = new FormData();
            formData.append('file', chunk.chunk);
            formData.append('hash', this.fileHash);
            formData.append('uploadId', this.uploadId);
            formData.append('chunkIndex', chunk.index);
            formData.append('totalChunks', this.chunks.length);
            formData.append('chunkSize', chunk.size);
            
            const response = await fetch('/api/upload/chunk', {
                method: 'POST',
                body: formData
            });
            
            if (!response.ok) {
                throw new Error(`上传失败: ${response.status}`);
            }
            
            const result = await response.json();
            
            this.uploadedChunks.add(chunk.index);
            this.uploadingChunks.delete(chunk.index);
            this.failedChunks.delete(chunk.index);
            
            // 更新进度
            this.onProgress({
                uploaded: this.uploadedChunks.size,
                total: this.chunks.length,
                percentage: (this.uploadedChunks.size / this.chunks.length * 100).toFixed(2)
            });
            
            return result;
            
        } catch (error) {
            this.uploadingChunks.delete(chunk.index);
            
            // 处理重试
            const failedCount = this.failedChunks.get(chunk.index) || 0;
            
            if (failedCount < this.retryAttempts) {
                this.failedChunks.set(chunk.index, failedCount + 1);
                console.warn(`分片 ${chunk.index} 上传失败，准备重试 (${failedCount + 1}/${this.retryAttempts})`);
                
                // 延迟重试
                setTimeout(() => {
                    if (!this.isPaused && this.isUploading) {
                        this.uploadSingleChunk(chunk);
                    }
                }, 1000 * Math.pow(2, failedCount)); // 指数退避
            } else {
                this.onError(new Error(`分片 ${chunk.index} 上传失败，已达到最大重试次数`));
                this.cancel();
            }
        }
    }
    
    // 合并文件
    async mergeFile() {
        const response = await fetch('/api/upload/merge', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                uploadId: this.uploadId,
                fileName: this.file.name,
                fileHash: this.fileHash,
                totalChunks: this.chunks.length
            })
        });
        
        if (!response.ok) {
            throw new Error('文件合并失败');
        }
        
        return await response.json();
    }
}
```

### 2. 使用示例
```javascript
// 使用断点续传
const fileInput = document.getElementById('file-input');
const progressBar = document.getElementById('progress-bar');
const progressText = document.getElementById('progress-text');

fileInput.addEventListener('change', async (event) => {
    const file = event.target.files[0];
    if (!file) return;
    
    const uploader = new ResumableUploader(file, {
        chunkSize: 2 * 1024 * 1024, // 2MB
        concurrentUploads: 5,
        retryAttempts: 3,
        onProgress: (progress) => {
            progressBar.style.width = `${progress.percentage}%`;
            progressText.textContent = `${progress.uploaded}/${progress.total} (${progress.percentage}%)`;
        },
        onComplete: () => {
            console.log('文件上传完成');
            progressText.textContent = '上传完成';
        },
        onError: (error) => {
            console.error('上传错误:', error);
            progressText.textContent = `上传失败: ${error.message}`;
        }
    });
    
    try {
        await uploader.init();
        await uploader.start();
    } catch (error) {
        console.error('初始化失败:', error);
    }
});

// 暂停/继续控制
const pauseBtn = document.getElementById('pause-btn');
const resumeBtn = document.getElementById('resume-btn');

let currentUploader = null;

pauseBtn.addEventListener('click', () => {
    if (currentUploader) {
        currentUploader.pause();
    }
});

resumeBtn.addEventListener('click', () => {
    if (currentUploader) {
        currentUploader.resume();
    }
});
```

```javascript
        this.worker.onmessage = (event) => {
```

### 2. 并发控制优化
```javascript
// 智能并发控制
class SmartConcurrentController {
    constructor(maxConcurrent = 5) {
        this.maxConcurrent = maxConcurrent;
        this.active = new Set();
        this.pending = [];
        this.successCount = 0;
        this.failureCount = 0;
        this.lastSpeed = 0;
    }
    
    async execute(task) {
        return new Promise((resolve, reject) => {
            const wrappedTask = async () => {
                const startTime = Date.now();
                
                try {
                    const result = await task();
                    this.successCount++;
                    this.updateSpeed(Date.now() - startTime);
                    resolve(result);
                } catch (error) {
                    this.failureCount++;
                    reject(error);
                } finally {
                    this.active.delete(wrappedTask);
                    this.processNext();
                }
            };
            
            this.pending.push(wrappedTask);
            this.processNext();
        });
    }
    
    processNext() {
        if (this.active.size >= this.maxConcurrent || this.pending.length === 0) {
            return;
        }
        
        // 动态调整并发数
        this.adjustConcurrency();
        
        const task = this.pending.shift();
        this.active.add(task);
        task();
    }
    
    adjustConcurrency() {
        const successRate = this.successCount / (this.successCount + this.failureCount);
        
        if (successRate > 0.95 && this.lastSpeed < 1000) {
            // 网络状况良好，增加并发数
            this.maxConcurrent = Math.min(this.maxConcurrent + 1, 10);
        } else if (successRate < 0.8 || this.lastSpeed > 5000) {
            // 网络状况较差，减少并发数
            this.maxConcurrent = Math.max(this.maxConcurrent - 1, 1);
        }
    }
    
    updateSpeed(duration) {
        this.lastSpeed = duration;
    }
}
```

### 1. 网络异常处理
```javascript
// 网络状态监控
class NetworkMonitor {
    constructor() {
        this.online = navigator.onLine;
        this.listeners = [];
        
        window.addEventListener('online', () => {
            this.online = true;
            this.notifyListeners('online');
        });
        
        window.addEventListener('offline', () => {
            this.online = false;
            this.notifyListeners('offline');
        });
    }
    
    addListener(callback) {
        this.listeners.push(callback);
    }
    
    notifyListeners(event) {
        this.listeners.forEach(callback => callback(event));
    }
    
    isOnline() {
        return this.online;
    }
}

// 集成网络监控
const networkMonitor = new NetworkMonitor();

networkMonitor.addListener((event) => {
    if (event === 'offline') {
        // 网络断开，暂停所有上传
        if (currentUploader) {
            currentUploader.pause();
        }
    } else if (event === 'online') {
        // 网络恢复，继续上传
        if (currentUploader) {
            currentUploader.resume();
        }
    }
});
```
