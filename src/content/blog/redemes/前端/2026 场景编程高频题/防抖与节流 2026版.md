## 基础概念回顾

### 防抖（Debounce）
防抖的核心思想：在事件被触发后，等待一段时间再执行回调。如果在这段时间内事件再次被触发，则重新计时。

**典型应用场景：**
- 搜索框输入（避免频繁请求）
- 窗口大小调整
- 表单验证

### 节流（Throttle）
节流的核心思想：在一定时间间隔内，只执行一次回调。

**典型应用场景：**
- 滚动事件处理
- 鼠标移动事件
- 按钮防重复点击

## 现代实现方案

### 1. 基础防抖实现（2026版）

```javascript
// 现代防抖函数（支持立即执行和取消）
function debounce(func, wait, immediate = false) {
    let timeoutId = null;
    let lastArgs = null;
    let lastThis = null;
    let result;

    const later = () => {
        timeoutId = null;
        if (!immediate) {
            result = func.apply(lastThis, lastArgs);
            if (!timeoutId) {
                lastArgs = lastThis = null;
            }
        }
    };

    const debounced = function(...args) {
        const callNow = immediate && !timeoutId;
        
        // 清除之前的定时器
        if (timeoutId) {
            clearTimeout(timeoutId);
        }

        lastArgs = args;
        lastThis = this;
        timeoutId = setTimeout(later, wait);

        if (callNow) {
            result = func.apply(lastThis, lastArgs);
        }

        return result;
    };

    // 添加取消功能
    debounced.cancel = () => {
        if (timeoutId) {
            clearTimeout(timeoutId);
            timeoutId = lastArgs = lastThis = null;
        }
    };

    // 添加立即执行功能
    debounced.flush = () => {
        if (timeoutId) {
            clearTimeout(timeoutId);
            later();
        }
    };

    return debounced;
}

// 使用示例
const searchInput = document.getElementById('search');

const handleSearch = debounce((query) => {
    console.log('搜索:', query);
    // 执行搜索逻辑
    performSearch(query);
}, 300);

searchInput.addEventListener('input', (e) => {
    handleSearch(e.target.value);
});
```

### 2. 基础节流实现（2026版）

```javascript
// 现代节流函数（支持首尾执行控制）
function throttle(func, wait, options = {}) {
    let timeoutId = null;
    let lastArgs = null;
    let lastThis = null;
    let result;
    let lastCallTime = 0;
    
    const { leading = true, trailing = true } = options;

    const invokeFunc = (time) => {
        lastCallTime = time;
        result = func.apply(lastThis, lastArgs);
        lastArgs = lastThis = null;
        return result;
    };

    const shouldInvoke = (time) => {
        const timeSinceLastCall = time - lastCallTime;
        return lastCallTime === 0 || timeSinceLastCall >= wait;
    };

    const remainingWait = (time) => {
        const timeSinceLastCall = time - lastCallTime;
        return wait - timeSinceLastCall;
    };

    const trailingEdge = (time) => {
        timeoutId = null;
        if (trailing && lastArgs) {
            return invokeFunc(time);
        }
        lastArgs = lastThis = null;
        return result;
    };

    const timerExpired = () => {
        const time = Date.now();
        if (shouldInvoke(time)) {
            return trailingEdge(time);
        }
        // 重新启动定时器
        timeoutId = setTimeout(timerExpired, remainingWait(time));
    };

    const throttled = function(...args) {
        const time = Date.now();
        const isInvoking = shouldInvoke(time);

        lastArgs = args;
        lastThis = this;

        if (isInvoking) {
            if (timeoutId === null) {
                if (leading) {
                    invokeFunc(time);
                }
                timeoutId = setTimeout(timerExpired, wait);
            } else if (trailing) {
                // 已经在等待中，不需要额外处理
            }
        }
        
        return result;
    };

    // 添加取消功能
    throttled.cancel = () => {
        if (timeoutId) {
            clearTimeout(timeoutId);
        }
        lastCallTime = 0;
        lastArgs = lastThis = timeoutId = null;
    };

    // 添加立即执行功能
    throttled.flush = () => {
        return timeoutId === null ? result : trailingEdge(Date.now());
    };

    return throttled;
}

// 使用示例
const handleScroll = throttle(() => {
    console.log('处理滚动事件');
    // 执行滚动相关逻辑
    updateScrollIndicator();
}, 100);

window.addEventListener('scroll', handleScroll);
```

function useDebounce(callback, delay, dependencies = []) {
    const callbackRef = useRef(callback);
    const timeoutRef = useRef();

    // 更新回调引用
    useEffect(() => {
        callbackRef.current = callback;
    }, [callback]);

    const debouncedCallback = useCallback((...args) => {
        if (timeoutRef.current) {
            clearTimeout(timeoutRef.current);
        }

        timeoutRef.current = setTimeout(() => {
            callbackRef.current(...args);
        }, delay);
    }, [delay, ...dependencies]);

    // 清理定时器
    useEffect(() => {
        return () => {
            if (timeoutRef.current) {
                clearTimeout(timeoutRef.current);
            }
        };
    }, []);

    return debouncedCallback;
}

// 自定义节流 Hook
function useThrottle(callback, delay, dependencies = []) {
    const callbackRef = useRef(callback);
    const lastRunRef = useRef(Date.now());

    useEffect(() => {
        callbackRef.current = callback;
    }, [callback]);

    const throttledCallback = useCallback((...args) => {
        const now = Date.now();
        
        if (now - lastRunRef.current >= delay) {
            lastRunRef.current = now;
            callbackRef.current(...args);
        }
    }, [delay, ...dependencies]);

    return throttledCallback;
}

        // 执行搜索

    }, 300);

    const handleInputChange = (e) => {
        const value = e.target.value;
        setQuery(value);
        handleSearch(value);
    };

    return (
        <input 
            value={query}
            onChange={handleInputChange}
            placeholder="搜索..."
        />
    );
}

### 5. 高性能防抖实现

```javascript
// 高性能防抖（支持批量处理）
class AdvancedDebouncer {
    constructor(wait, maxWait = null) {
        this.wait = wait;
        this.maxWait = maxWait;
        this.timeoutId = null;
        this.maxTimeoutId = null;
        this.lastArgs = [];
        this.lastThis = null;
        this.lastCallTime = 0;
        this.lastInvokeTime = 0;
    }

    invokeFunc(time) {
        const args = this.lastArgs;
        const thisArg = this.lastThis;
        
        this.lastArgs = this.lastThis = null;
        this.lastInvokeTime = time;
        
        // 批量处理参数
        if (args.length > 0 && Array.isArray(args[0])) {
            return this.batchProcess(args[0]);
        }
        
        return this.func.apply(thisArg, args);
    }

    batchProcess(batchArgs) {
        // 批量处理逻辑
        console.log('批量处理参数:', batchArgs.length);
        return batchArgs.map(arg => this.func.apply(this.lastThis, [arg]));
    }

    shouldInvoke(time) {
        const timeSinceLastCall = time - this.lastCallTime;
        const timeSinceLastInvoke = time - this.lastInvokeTime;
        
        return (this.lastCallTime === 0) || 
               (timeSinceLastCall >= this.wait) ||
               (this.maxWait !== null && timeSinceLastInvoke >= this.maxWait);
    }

    trailingEdge(time) {
        this.timeoutId = null;
        
        if (this.lastArgs.length > 0) {
            return this.invokeFunc(time);
        }
        
        this.lastArgs = this.lastThis = null;
        return null;
    }

    debounce(func) {
        this.func = func;
        
        const debounced = (...args) => {
            const time = Date.now();
            const isInvoking = this.shouldInvoke(time);
            
            this.lastArgs.push(args);
            this.lastThis = this;
            this.lastCallTime = time;
            
            if (isInvoking) {
                if (this.timeoutId === null) {
                    this.lastInvokeTime = time;
                    this.timeoutId = setTimeout(() => {
                        this.trailingEdge(Date.now());
                    }, this.wait);
                    
                    if (this.maxWait !== null) {
                        this.maxTimeoutId = setTimeout(() => {
                            this.trailingEdge(Date.now());
                        }, this.maxWait);
                    }
                }
            }
            
            return null;
        };
        
        debounced.cancel = () => {
            if (this.timeoutId !== null) {
                clearTimeout(this.timeoutId);
            }
            if (this.maxTimeoutId !== null) {
                clearTimeout(this.maxTimeoutId);
            }
            
            this.lastCallTime = 0;
            this.lastArgs = [];
            this.lastThis = null;
            this.timeoutId = this.maxTimeoutId = null;
        };
        
        return debounced;
    }
}

// 使用示例
const debouncer = new AdvancedDebouncer(300, 1000);
const optimizedSearch = debouncer.debounce((query) => {
    console.log('优化搜索:', query);
});
```

### 6. 自适应节流（基于性能）

```javascript
// 自适应节流（根据设备性能调整间隔）
class AdaptiveThrottler {
    constructor(baseInterval = 100) {
        this.baseInterval = baseInterval;
        this.performanceScore = this.calculatePerformanceScore();
        this.currentInterval = this.calculateOptimalInterval();
        this.lastExecution = 0;
        this.executionHistory = [];
    }

    // 计算设备性能分数
    calculatePerformanceScore() {
        let score = 1.0;
        
        // 基于硬件并发数
        if (navigator.hardwareConcurrency) {
            score *= Math.min(navigator.hardwareConcurrency / 4, 2);
        }
        
        // 基于内存（如果可用）
        if (performance.memory) {
            const memoryGB = performance.memory.totalJSHeapSize / (1024 * 1024 * 1024);
            score *= Math.min(memoryGB / 4, 1.5);
        }
        
        // 基于帧率（简单检测）
        score *= this.detectFrameRateFactor();
        
        return Math.max(0.5, Math.min(score, 3.0));
    }

    // 检测帧率因素
    detectFrameRateFactor() {
        let frameCount = 0;
        let startTime = performance.now();
        
        const checkFrame = () => {
            frameCount++;
            
            if (performance.now() - startTime < 1000) {
                requestAnimationFrame(checkFrame);
            } else {
                const fps = frameCount;
                return Math.min(fps / 60, 1.5);
            }
        };
        
        requestAnimationFrame(checkFrame);
        return 1.0; // 默认值
    }

    // 计算最优间隔
    calculateOptimalInterval() {
        return this.baseInterval / this.performanceScore;
    }

    // 自适应节流函数
    throttle(func) {
        return (...args) => {
            const now = performance.now();
            
            // 记录执行历史
            this.executionHistory.push(now);
            if (this.executionHistory.length > 10) {
                this.executionHistory.shift();
            }
            
            // 动态调整间隔
            if (this.executionHistory.length >= 5) {
                const recentInterval = this.executionHistory[4] - this.executionHistory[0];
                const averageInterval = recentInterval / 4;
                
                if (averageInterval < this.currentInterval * 0.8) {
                    // 执行过于频繁，增加间隔
                    this.currentInterval = Math.min(this.currentInterval * 1.2, this.baseInterval * 2);
                } else if (averageInterval > this.currentInterval * 1.5) {
                    // 执行间隔过大，减少间隔
                    this.currentInterval = Math.max(this.currentInterval * 0.8, this.baseInterval / 3);
                }
            }
            
            if (now - this.lastExecution >= this.currentInterval) {
                this.lastExecution = now;
                return func.apply(this, args);
            }
        };
    }
}

// 使用示例
const adaptiveThrottler = new AdaptiveThrottler(100);
const adaptiveScrollHandler = adaptiveThrottler.throttle(() => {
    console.log('自适应滚动处理');
});

window.addEventListener('scroll', adaptiveScrollHandler);
```

### 7. 搜索框智能防抖

```javascript
// 智能搜索防抖（根据输入速度调整延迟）
class SmartSearchDebouncer {
    constructor() {
        this.baseDelay = 300;
        this.minDelay = 100;
        this.maxDelay = 1000;
        this.typingSpeedHistory = [];
        this.lastTypingTime = 0;
        this.currentDelay = this.baseDelay;
    }

    // 根据输入速度计算延迟
    calculateDynamicDelay() {
        if (this.typingSpeedHistory.length < 3) {
            return this.baseDelay;
        }
        
        const recentSpeeds = this.typingSpeedHistory.slice(-3);
        const averageSpeed = recentSpeeds.reduce((a, b) => a + b, 0) / recentSpeeds.length;
        
        // 输入越快，延迟越短
        if (averageSpeed > 200) { // 快速输入
            return this.minDelay;
        } else if (averageSpeed < 50) { // 慢速输入
            return this.maxDelay;
        } else {
            // 线性映射
            const ratio = (averageSpeed - 50) / 150; // 50-200ms 映射到 maxDelay-minDelay
            return this.maxDelay - ratio * (this.maxDelay - this.minDelay);
        }
    }

    // 记录输入速度
    recordTypingSpeed() {
        const now = Date.now();
        
        if (this.lastTypingTime > 0) {
            const speed = now - this.lastTypingTime;
            this.typingSpeedHistory.push(speed);
            
            // 保持最近10次记录
            if (this.typingSpeedHistory.length > 10) {
                this.typingSpeedHistory.shift();
            }
        }
        
        this.lastTypingTime = now;
        this.currentDelay = this.calculateDynamicDelay();
    }

    // 智能防抖函数
    smartDebounce(func) {
        let timeoutId = null;
        
        return (...args) => {
            this.recordTypingSpeed();
            
            if (timeoutId) {
                clearTimeout(timeoutId);
            }
            
            timeoutId = setTimeout(() => {
                func.apply(this, args);
            }, this.currentDelay);
        };
    }
}

// 使用示例
const smartDebouncer = new SmartSearchDebouncer();
const smartSearch = smartDebouncer.smartDebounce((query) => {
    console.log(`智能搜索 [延迟:${smartDebouncer.currentDelay}ms]:`, query);
    performSearch(query);
});

document.getElementById('search').addEventListener('input', (e) => {
    smartSearch(e.target.value);
});

// 滚动加载优化节流
class ScrollLoadThrottler {
    constructor(loadThreshold = 100) {
        this.loadThreshold = loadThreshold;
        this.lastScrollY = 0;
        this.lastScrollTime = 0;
        this.pendingLoad = false;
        this.scrollDirection = 'down';
    }

    // 智能滚动处理
    handleScroll() {
        const currentScrollY = window.scrollY;
        const currentTime = Date.now();
        
        // 判断滚动方向
        this.scrollDirection = currentScrollY > this.lastScrollY ? 'down' : 'up';
        
        // 只有向下滚动且接近底部时才触发加载
        const nearBottom = this.isNearBottom(currentScrollY);
        
        if (this.scrollDirection === 'down' && nearBottom && !this.pendingLoad) {
            this.pendingLoad = true;
            
            // 使用 requestAnimationFrame 确保在重绘前执行
            requestAnimationFrame(() => {
                this.loadMoreContent();
                this.pendingLoad = false;
            });
        }
        
        this.lastScrollY = currentScrollY;
        this.lastScrollTime = currentTime;
    }

    // 判断是否接近底部
    isNearBottom(scrollY) {
        const windowHeight = window.innerHeight;
        const documentHeight = document.documentElement.scrollHeight;
        return scrollY + windowHeight >= documentHeight - this.loadThreshold;
    }

    // 加载更多内容
    loadMoreContent() {
        console.log('加载更多内容...');
        // 执行加载逻辑
        this.fetchMoreData()
            .then(data => {
                this.appendContent(data);
            })
            .catch(error => {
                console.error('加载失败:', error);
            });
    }
}

// 使用示例
const scrollThrottler = new ScrollLoadThrottler(200);
const optimizedScrollHandler = throttle(
    scrollThrottler.handleScroll.bind(scrollThrottler), 
    50
);

window.addEventListener('scroll', optimizedScrollHandler);
```

class DebounceThrottleTester {

                debouncedTest();

## 2026年最佳实践

1. **选择合适的策略**：根据具体场景选择防抖或节流
2. **考虑设备性能**：在高性能设备上可以使用更短的间隔
3. **动态调整参数**：根据用户行为动态调整防抖/节流参数
4. **错误处理**：确保在极端情况下也能正常工作
5. **性能监控**：在生产环境中监控防抖节流的效果
6. **渐进增强**：为不支持某些API的浏览器提供降级方案

通过合理使用防抖和节流技术，可以显著提升前端应用的性能和用户体验。在2026年的前端开发中，这些技术依然是优化高频事件处理的必备工具。
