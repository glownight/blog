## 一、内容安全策略（CSP）

### 1. CSP 3.0 高级配置

```javascript
// CSP 配置管理器
class CSPManager {
    constructor() {
        this.directives = new Map();
        this.reportOnly = false;
        this.reportUri = null;
    }

    // 设置默认策略
    setDefaultPolicy() {
        this.directives.set('default-src', ["'self'"]);
        this.directives.set('script-src', ["'self'", "'unsafe-inline'", "https://cdn.example.com"]);
        this.directives.set('style-src', ["'self'", "'unsafe-inline'"]);
        this.directives.set('img-src', ["'self'", "data:", "https:"]);
        this.directives.set('connect-src', ["'self'"]);
        this.directives.set('font-src', ["'self'"]);
        this.directives.set('object-src', ["'none'"]);
        this.directives.set('media-src', ["'self'"]);
        this.directives.set('frame-src', ["'self'"]);
        
        // CSP 3.0 新增指令
        this.directives.set('worker-src', ["'self'"]);
        this.directives.set('manifest-src', ["'self'"]);
        this.directives.set('prefetch-src', ["'self'"]);
        this.directives.set('navigate-to', ["'self'"]);
    }

    // 添加可信域名
    addTrustedDomain(directive, domain) {
        if (!this.directives.has(directive)) {
            this.directives.set(directive, []);
        }
        
        const sources = this.directives.get(directive);
        if (!sources.includes(domain)) {
            sources.push(domain);
        }
    }

    // 生成 CSP 头部
    generateHeader() {
        const headerParts = [];
        
        for (const [directive, sources] of this.directives.entries()) {
            headerParts.push(`${directive} ${sources.join(' ')}`);
        }
        
        if (this.reportUri) {
            headerParts.push(`report-uri ${this.reportUri}`);
            headerParts.push(`report-to ${this.reportUri.replace('uri', 'to')}`);
        }
        
        const headerValue = headerParts.join('; ');
        
        if (this.reportOnly) {
            return `Content-Security-Policy-Report-Only: ${headerValue}`;
        } else {
            return `Content-Security-Policy: ${headerValue}`;
        }
    }

    // 验证策略有效性
    validatePolicy() {
        const warnings = [];
        
        // 检查是否存在不安全指令
        if (this.directives.get('script-src')?.includes("'unsafe-eval'")) {
            warnings.push('检测到 unsafe-eval，存在安全风险');
        }
        
        if (this.directives.get('script-src')?.includes("'unsafe-inline'")) {
            warnings.push('检测到 unsafe-inline，建议使用 nonce 或 hash');
        }
        
        // 检查对象源
        if (this.directives.get('object-src')?.includes("'self'")) {
            warnings.push('object-src 包含 self，建议设置为 none');
        }
        
        return warnings;
    }

    // 动态 nonce 生成
    generateNonce() {
        const nonce = crypto.getRandomValues(new Uint8Array(16));
        return Array.from(nonce).map(b => b.toString(16).padStart(2, '0')).join('');
    }

    // 为内联脚本添加 nonce
    addNonceToScript(nonce) {
        this.directives.set('script-src', [
            ...(this.directives.get('script-src') || []),
            `'nonce-${nonce}'`
        ]);
    }
}

class CSPReportHandler {
    constructor() {
        this.reports = [];
        this.threshold = 10; // 阈值触发警报
    }

    // 处理 CSP 违规报告
    async handleReport(reportData) {
        const report = {
            id: this.generateReportId(),
            timestamp: new Date().toISOString(),
            data: reportData,
            severity: this.calculateSeverity(reportData)
        };

        this.reports.push(report);
        
        // 检查是否需要触发警报
        if (this.shouldTriggerAlert(report)) {
            await this.triggerAlert(report);
        }
        
        return report;
    }

    // 计算违规严重性
    calculateSeverity(reportData) {
        const blockedUri = reportData['blocked-uri'];
        const violatedDirective = reportData['violated-directive'];
        
        let severity = 'low';
        
        if (violatedDirective.includes('script-src')) {
            severity = 'high';
        } else if (violatedDirective.includes('connect-src')) {
            severity = 'medium';
        }
        
        return severity;
    }

    // 检查是否触发警报
    shouldTriggerAlert(report) {
        // 高严重性违规立即触发
        if (report.severity === 'high') {
            return true;

        // 检查最近的高频违规
        const recentReports = this.reports.filter(r => 
            Date.now() - new Date(r.timestamp).getTime() < 5 * 60 * 1000 // 5分钟内
        );
        
        return recentReports.length >= this.threshold;
```

### 2. 动态 CSP 策略
```javascript
// 动态 CSP 策略管理器
class DynamicCSPManager {
    constructor() {
        this.basePolicy = new CSPManager();
        this.dynamicRules = new Map();
        this.ruleEngine = new CSPRuleEngine();
    }

    // 根据页面类型生成策略
    generatePolicyForPage(pageType, userContext) {
        let policy = this.basePolicy.generateHeader();
        
        // 应用动态规则
        const dynamicRules = this.ruleEngine.evaluateRules(pageType, userContext);
        
        for (const rule of dynamicRules) {
            policy = this.applyRule(policy, rule);
        }
        
        return policy;
    }

    // 应用规则到策略
    applyRule(policy, rule) {
        // 根据规则类型修改策略
        switch (rule.type) {
            case 'add_domain':
                return this.addDomainToPolicy(policy, rule.directive, rule.domain);
            case 'remove_domain':
                return this.removeDomainFromPolicy(policy, rule.directive, rule.domain);
            case 'set_nonce':
                return this.setNonceInPolicy(policy, rule.nonce);
            default:
                return policy;
        }
    }

    // 添加域名到策略
    addDomainToPolicy(policy, directive, domain) {
        // 解析现有策略
        const policyParts = policy.split(';').map(part => part.trim());
        
        // 找到对应的指令
        const directiveIndex = policyParts.findIndex(part => part.startsWith(directive));
        
        if (directiveIndex !== -1) {
            // 添加域名
            policyParts[directiveIndex] = policyParts[directiveIndex] + ' ' + domain;
        } else {
            // 创建新指令
            policyParts.push(`${directive} ${domain}`);
        }
        
        return policyParts.join('; ');
    }

    // 设置 nonce
    setNonceInPolicy(policy, nonce) {
        return policy.replace(
            /script-src[^;]*/,
            `script-src 'nonce-${nonce}'`
        );
    }
}

// CSP 规则引擎
class CSPRuleEngine {
    constructor() {
        this.rules = [
            {
                condition: (pageType, userContext) => pageType === 'admin',
                action: () => [
                    { type: 'add_domain', directive: 'script-src', domain: 'https://admin-cdn.example.com' }
                ]
            },
            {
                condition: (pageType, userContext) => userContext.trustLevel === 'high',
                action: () => [
                    { type: 'add_domain', directive: 'connect-src', domain: 'wss://realtime.example.com' }
                ]
            }
        ];
    }

    // 评估规则
    evaluateRules(pageType, userContext) {
        const applicableRules = [];
        
        for (const rule of this.rules) {
            if (rule.condition(pageType, userContext)) {
                applicableRules.push(...rule.action());
            }
        }
        
        return applicableRules;
    }
}
```

```javascript
// CORS 配置管理器
class CORSManager {
    constructor() {
        this.allowedOrigins = new Set();
        this.allowedMethods = new Set(['GET', 'POST', 'PUT', 'DELETE']);
        this.allowedHeaders = new Set(['Content-Type', 'Authorization']);
        this.exposedHeaders = new Set();
        this.maxAge = 86400; // 24小时
        this.credentials = false;
    }

    // 添加允许的源
    addAllowedOrigin(origin) {
        // 验证 origin 格式
        if (this.isValidOrigin(origin)) {
            this.allowedOrigins.add(origin);
        } else {
            throw new Error(`无效的 origin: ${origin}`);
        }
    }

    // 验证 origin 格式
    isValidOrigin(origin) {
        if (origin === '*') return true;
        
        try {
            const url = new URL(origin);
            return url.protocol === 'http:' || url.protocol === 'https:';
        } catch {
            return false;
        }
    }

    // 生成 CORS 头部
    generateHeaders(requestOrigin) {
        const headers = {};
        
        // 检查 origin 是否允许
        if (this.isOriginAllowed(requestOrigin)) {
            headers['Access-Control-Allow-Origin'] = requestOrigin;
            
            if (this.credentials) {
                headers['Access-Control-Allow-Credentials'] = 'true';
            }
        } else {
            headers['Access-Control-Allow-Origin'] = 'null';
        }
        
        // 预检请求处理
        if (this.isPreflightRequest()) {
            headers['Access-Control-Allow-Methods'] = Array.from(this.allowedMethods).join(', ');
            headers['Access-Control-Allow-Headers'] = Array.from(this.allowedHeaders).join(', ');
            headers['Access-Control-Max-Age'] = this.maxAge.toString();
        }
        
        // 暴露头部
        if (this.exposedHeaders.size > 0) {
            headers['Access-Control-Expose-Headers'] = Array.from(this.exposedHeaders).join(', ');
        }
        
        return headers;
    }

    // 检查 origin 是否允许
    isOriginAllowed(origin) {
        if (this.allowedOrigins.has('*')) {
            return true;
        }
        
        return this.allowedOrigins.has(origin);
    }

    // 检查是否为预检请求
    isPreflightRequest() {
        // 在实际实现中会检查请求方法
        return false;
    }

    // 验证 CORS 请求
    validateRequest(request) {
        const violations = [];
        
        // 检查 origin
        if (!this.isOriginAllowed(request.origin)) {
            violations.push('Origin not allowed');
        }
        
        // 检查方法
        if (!this.allowedMethods.has(request.method)) {
            violations.push('Method not allowed');
        }
        
        // 检查头部
        for (const header of request.headers) {
            if (!this.allowedHeaders.has(header)) {
                violations.push(`Header not allowed: ${header}`);
            }
        }
        
        return violations;
    }
}

// CORS 安全监控
class CORSSecurityMonitor {
    constructor() {
        this.suspiciousRequests = [];
        this.patterns = this.getSuspiciousPatterns();
    }

    // 监控 CORS 请求
    monitorRequest(request) {
        const riskScore = this.calculateRiskScore(request);
        
        if (riskScore > 0.7) {
            this.recordSuspiciousRequest(request, riskScore);
            
            if (riskScore > 0.9) {
                this.triggerSecurityAlert(request);
            }
        }
        
        return riskScore;
    }

    // 计算风险分数
    calculateRiskScore(request) {
        let score = 0;
        
        // 检查可疑 origin
        if (this.isSuspiciousOrigin(request.origin)) {
            score += 0.3;
        }
        
        // 检查可疑方法
        if (this.isSuspiciousMethod(request.method)) {
            score += 0.2;
        }
        
        // 检查可疑头部
        if (this.hasSuspiciousHeaders(request.headers)) {
            score += 0.3;
        }
        
        // 检查频率
        if (this.isHighFrequency(request)) {
            score += 0.2;
        }
        
        return Math.min(score, 1.0);
    }

    // 获取可疑模式
    getSuspiciousPatterns() {
        return {
            origins: [
                /^http:\/\/localhost/,
                /^http:\/\/127\.0\.0\.1/,
                /^http:\/\/192\.168\./,
                /^http:\/\/10\./,
                /^http:\/\/172\.(1[6-9]|2[0-9]|3[0-1])\./
            ],
            methods: ['TRACE', 'TRACK', 'CONNECT'],
            headers: ['X-Forwarded-For', 'X-Real-IP', 'X-Originating-IP']
        };
    }

    isSuspiciousOrigin(origin) {
        return this.patterns.origins.some(pattern => pattern.test(origin));
    }

    isSuspiciousMethod(method) {
        return this.patterns.methods.includes(method.toUpperCase());
    }

    hasSuspiciousHeaders(headers) {
        return headers.some(header => 
            this.patterns.headers.includes(header)
        );
    }

    isHighFrequency(request) {
        // 检查请求频率
        const recentRequests = this.suspiciousRequests.filter(req => 
            Date.now() - req.timestamp < 60 * 1000 // 1分钟内
        );
        
        return recentRequests.length > 10;
    }

    recordSuspiciousRequest(request, riskScore) {
        this.suspiciousRequests.push({
            ...request,
            timestamp: Date.now(),
            riskScore
        });
    }

    triggerSecurityAlert(request) {
        console.warn('CORS 安全警报:', {
            type: 'suspicious_cors_request',
            request,
            timestamp: new Date().toISOString()
        });
    }
}
```

```javascript
// XSS 过滤器
class XSSFilter {
    constructor() {
        this.patterns = this.getXSSPatterns();
        this.sanitizer = new DOMPurify();
    }

    // 检测 XSS 攻击
    detectXSS(input) {
        const threats = [];
        
        // 检查脚本标签
        if (this.containsScriptTags(input)) {
            threats.push('script_tags');
        }
        
        // 检查事件处理器
        if (this.containsEventHandlers(input)) {
            threats.push('event_handlers');
        }
        
        // 检查 JavaScript 协议
        if (this.containsJavaScriptProtocol(input)) {
            threats.push('javascript_protocol');
        }
        
        // 检查数据 URI
        if (this.containsDataURI(input)) {
            threats.push('data_uri');
        }
        
        return threats;
    }

    // 清理 HTML
    sanitizeHTML(html) {
        return this.sanitizer.sanitize(html, {
            ALLOWED_TAGS: ['div', 'span', 'p', 'br', 'strong', 'em', 'ul', 'ol', 'li'],
            ALLOWED_ATTR: ['class', 'style'],
            FORBID_ATTR: ['onclick', 'onload', 'onerror']
        });
    }

    // 清理 URL
    sanitizeURL(url) {
        try {
            const parsed = new URL(url);
            
            // 只允许 http 和 https
            if (parsed.protocol !== 'http:' && parsed.protocol !== 'https:') {
                return null;
            }
            
            // 清理查询参数
            const cleanParams = new URLSearchParams();
            for (const [key, value] of parsed.searchParams.entries()) {
                if (this.isSafeParameter(key, value)) {
                    cleanParams.set(key, value);
                }
            }
            
            parsed.search = cleanParams.toString();
            return parsed.toString();
            
        } catch {
            return null;
        }
    }

    // 获取 XSS 模式
    getXSSPatterns() {
        return {
            scriptTags: /<script[^>]*>.*?<\/script>/gi,
            eventHandlers: /on\w+\s*=/gi,
            javascriptProtocol: /javascript:/gi,
            dataURI: /data:/gi,
            iframe: /<iframe[^>]*>.*?<\/iframe>/gi,
            object: /<object[^>]*>.*?<\/object>/gi,
            embed: /<embed[^>]*>.*?<\/embed>/gi
        };
    }

    containsScriptTags(input) {
        return this.patterns.scriptTags.test(input);
    }

    containsEventHandlers(input) {
        return this.patterns.eventHandlers.test(input);
    }

    containsJavaScriptProtocol(input) {
        return this.patterns.javascriptProtocol.test(input);
    }

    containsDataURI(input) {
        return this.patterns.dataURI.test(input);
    }

    isSafeParameter(key, value) {
        // 检查参数是否安全
        const unsafePatterns = [
            /<script/i,
            /javascript:/i,
            /vbscript:/i,
            /data:/i
        ];
        
        return !unsafePatterns.some(pattern => 
            pattern.test(key) || pattern.test(value)
        );
    }
}

// 基于内容的 XSS 防护
class ContentBasedXSSProtection {
    constructor() {
        this.validators = {
            html: new HTMLValidator(),
            css: new CSSValidator(),
            javascript: new JavaScriptValidator(),
            url: new URLValidator()
        };
    }

    // 根据内容类型验证输入
    validateInput(input, contentType) {
        const validator = this.validators[contentType];
        
        if (!validator) {
            throw new Error(`不支持的内容类型: ${contentType}`);
        }
        
        return validator.validate(input);
    }

    // 自动检测内容类型并验证
    autoValidate(input) {
        // 检测内容类型
        const contentType = this.detectContentType(input);
        
        // 使用对应的验证器
        return this.validateInput(input, contentType);
    }

    detectContentType(input) {
        if (input.trim().startsWith('<')) {
            return 'html';
        } else if (input.includes('://')) {
            return 'url';
        } else if (input.includes('{') && input.includes('}')) {
            return 'css';
        } else {
            return 'text';
        }
    }
}

// HTML 验证器
class HTMLValidator {
    validate(html) {
        const threats = [];
        
        // 检查危险标签
        const dangerousTags = ['script', 'iframe', 'object', 'embed', 'base'];
        for (const tag of dangerousTags) {
            if (html.includes(`<${tag}`)) {
                threats.push(`dangerous_tag_${tag}`);
            }
        }
        
        // 检查危险属性
        const dangerousAttrs = ['onclick', 'onload', 'onerror', 'src', 'href'];
        for (const attr of dangerousAttrs) {
            if (html.includes(`${attr}=`)) {
                threats.push(`dangerous_attr_${attr}`);
            }
        }
        
        return {
            isValid: threats.length === 0,
            threats,
            sanitized: this.sanitize(html)
        };
    }

    sanitize(html) {
        // 使用 DOMPurify 或其他库进行清理
        return html.replace(/<script[^>]*>.*?<\/script>/gi, '')
                  .replace(/on\w+\s*=\s*["'][^"']*["']/gi, '');
    }
}

// URL 验证器
class URLValidator {
    validate(url) {
        try {
            const parsed = new URL(url);
            
            // 检查协议
            if (!['http:', 'https:'].includes(parsed.protocol)) {
                return {
                    isValid: false,
                    threats: ['unsafe_protocol'],
                    sanitized: null
                };
            }
            
            // 检查主机名
            if (this.isSuspiciousHostname(parsed.hostname)) {
                return {
                    isValid: false,
                    threats: ['suspicious_hostname'],
                    sanitized: null
                };
            }
            
            return {
                isValid: true,
                threats: [],
                sanitized: url
            };
            
        } catch {
            return {
                isValid: false,
                threats: ['invalid_url'],
                sanitized: null
            };
        }
    }

    isSuspiciousHostname(hostname) {
        const suspiciousPatterns = [
            /localhost/i,
            /127\.0\.0\.1/,
            /192\.168\./,
            /10\./,
            /172\.(1[6-9]|2[0-9]|3[0-1])\./
        ];
        
        return suspiciousPatterns.some(pattern => pattern.test(hostname));
    }
}
```
