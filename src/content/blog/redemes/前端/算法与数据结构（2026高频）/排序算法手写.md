## 一、基础排序算法

### 1. 冒泡排序
```javascript
// 冒泡排序 - 时间复杂度 O(n²)
function bubbleSort(arr) {
    const n = arr.length;
    
    for (let i = 0; i < n - 1; i++) {
        // 标记是否发生交换
        let swapped = false;
        
        for (let j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                // 交换元素
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
                swapped = true;
            }
        }
        
        // 如果没有发生交换，说明已经有序
        if (!swapped) break;
    }
    
    return arr;
}

// 优化版本 - 记录最后交换位置
function bubbleSortOptimized(arr) {
    let n = arr.length;
    let lastSwapIndex = n - 1;
    
    while (lastSwapIndex > 0) {
        let newLastSwapIndex = 0;
        
        for (let i = 0; i < lastSwapIndex; i++) {
            if (arr[i] > arr[i + 1]) {
                [arr[i], arr[i + 1]] = [arr[i + 1], arr[i]];
                newLastSwapIndex = i;
            }
        }
        
        lastSwapIndex = newLastSwapIndex;
    }
    
    return arr;
}
```

### 2. 选择排序
```javascript
// 选择排序 - 时间复杂度 O(n²)
function selectionSort(arr) {
    const n = arr.length;
    
    for (let i = 0; i < n - 1; i++) {
        // 找到最小元素的索引
        let minIndex = i;
        
        for (let j = i + 1; j < n; j++) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j;
            }
        }
        
        // 将最小元素交换到当前位置
        if (minIndex !== i) {
            [arr[i], arr[minIndex]] = [arr[minIndex], arr[i]];
        }
    }
    
    return arr;
}
```

### 3. 插入排序
```javascript
// 插入排序 - 时间复杂度 O(n²)
function insertionSort(arr) {
    const n = arr.length;
    
    for (let i = 1; i < n; i++) {
        const current = arr[i];
        let j = i - 1;
        
        // 将比当前元素大的元素后移
        while (j >= 0 && arr[j] > current) {
            arr[j + 1] = arr[j];
            j--;
        }
        
        // 插入当前元素
        arr[j + 1] = current;
    }
    
    return arr;
}

// 二分插入排序优化
function binaryInsertionSort(arr) {
    const n = arr.length;
    
    for (let i = 1; i < n; i++) {
        const current = arr[i];
        
        // 使用二分查找找到插入位置
        let left = 0;
        let right = i - 1;
        
        while (left <= right) {
            const mid = Math.floor((left + right) / 2);
            
            if (arr[mid] > current) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        
        // 将元素后移
        for (let j = i; j > left; j--) {
            arr[j] = arr[j - 1];
        }
        
        // 插入元素
        arr[left] = current;
    }
    
    return arr;
}
```

## 二、高级排序算法

### 1. 归并排序
```javascript
// 归并排序 - 时间复杂度 O(n log n)
function mergeSort(arr) {
    if (arr.length <= 1) {
        return arr;
    }
    
    // 分割数组
    const mid = Math.floor(arr.length / 2);
    const left = arr.slice(0, mid);
    const right = arr.slice(mid);
    
    // 递归排序并合并
    return merge(mergeSort(left), mergeSort(right));
}

// 合并两个有序数组
function merge(left, right) {
    const result = [];
    let i = 0, j = 0;
    
    while (i < left.length && j < right.length) {
        if (left[i] <= right[j]) {
            result.push(left[i]);
            i++;
        } else {
            result.push(right[j]);
            j++;
        }
    }
    
    // 添加剩余元素
    return result.concat(left.slice(i)).concat(right.slice(j));
}

// 原地归并排序（节省空间）
function mergeSortInPlace(arr, left = 0, right = arr.length - 1) {
    if (left >= right) return;
    
    const mid = Math.floor((left + right) / 2);
    
    mergeSortInPlace(arr, left, mid);
    mergeSortInPlace(arr, mid + 1, right);
    
    mergeInPlace(arr, left, mid, right);
}

function mergeInPlace(arr, left, mid, right) {
    const temp = [];
    let i = left, j = mid + 1;
    
    while (i <= mid && j <= right) {
        if (arr[i] <= arr[j]) {
            temp.push(arr[i]);
            i++;
        } else {
            temp.push(arr[j]);
            j++;
        }
    }
    
    // 复制剩余元素
    while (i <= mid) {
        temp.push(arr[i]);
        i++;
    }
    
    while (j <= right) {
        temp.push(arr[j]);
        j++;
    }
    
    // 将临时数组复制回原数组
    for (let k = 0; k < temp.length; k++) {
        arr[left + k] = temp[k];
    }
}
```

### 2. 快速排序
```javascript
// 快速排序 - 时间复杂度 O(n log n)
function quickSort(arr, left = 0, right = arr.length - 1) {
    if (left < right) {
        // 分区操作
        const pivotIndex = partition(arr, left, right);
        
        // 递归排序左右分区
        quickSort(arr, left, pivotIndex - 1);
        quickSort(arr, pivotIndex + 1, right);
    }
    
    return arr;
}

// 分区函数
function partition(arr, left, right) {
    // 选择基准元素（这里选择最后一个元素）
    const pivot = arr[right];
    let i = left;
    
    for (let j = left; j < right; j++) {
        if (arr[j] <= pivot) {
            // 将小于基准的元素交换到左侧
            [arr[i], arr[j]] = [arr[j], arr[i]];
            i++;
        }
    }
    
    // 将基准元素放到正确位置
    [arr[i], arr[right]] = [arr[right], arr[i]];
    
    return i;
}

// 三路快速排序（处理重复元素）
function quickSort3Way(arr, left = 0, right = arr.length - 1) {
    if (left >= right) return;
    
    // 三路分区
    let lt = left;      // arr[left...lt-1] < pivot
    let gt = right;     // arr[gt+1...right] > pivot
    let i = left + 1;   // arr[lt...i-1] == pivot
    
    const pivot = arr[left];
    
    while (i <= gt) {
        if (arr[i] < pivot) {
            [arr[i], arr[lt]] = [arr[lt], arr[i]];
            lt++;
            i++;
        } else if (arr[i] > pivot) {
            [arr[i], arr[gt]] = [arr[gt], arr[i]];
            gt--;
        } else {
            i++;
        }
    }
    
    quickSort3Way(arr, left, lt - 1);
    quickSort3Way(arr, gt + 1, right);
}
```

### 3. 堆排序
```javascript
// 堆排序 - 时间复杂度 O(n log n)
function heapSort(arr) {
    const n = arr.length;
    
    // 构建最大堆
    for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
        heapify(arr, n, i);
    }
    
    // 逐个提取最大元素
    for (let i = n - 1; i > 0; i--) {
        // 将最大元素移到末尾
        [arr[0], arr[i]] = [arr[i], arr[0]];
        
        // 重新堆化剩余元素
        heapify(arr, i, 0);
    }
    
    return arr;
}

// 堆化函数
function heapify(arr, n, i) {
    let largest = i;        // 初始化最大值为根节点
    const left = 2 * i + 1;  // 左子节点
    const right = 2 * i + 2; // 右子节点
    
    // 如果左子节点大于根节点
    if (left < n && arr[left] > arr[largest]) {
        largest = left;
    }
    
    // 如果右子节点大于当前最大值
    if (right < n && arr[right] > arr[largest]) {
        largest = right;
    }
    
    // 如果最大值不是根节点
    if (largest !== i) {
        [arr[i], arr[largest]] = [arr[largest], arr[i]];
        
        // 递归堆化受影响的子树
        heapify(arr, n, largest);
    }
}
```

## 三、特殊场景排序

### 1. 计数排序（适用于小范围整数）
```javascript
// 计数排序 - 时间复杂度 O(n + k)
function countingSort(arr) {
    if (arr.length === 0) return arr;
    
    // 找到最大值和最小值
    const max = Math.max(...arr);
    const min = Math.min(...arr);
    const range = max - min + 1;
    
    // 创建计数数组
    const count = new Array(range).fill(0);
    
    // 统计每个元素出现次数
    for (let i = 0; i < arr.length; i++) {
        count[arr[i] - min]++;
    }
    
    // 计算累计次数
    for (let i = 1; i < range; i++) {
        count[i] += count[i - 1];
    }
    
    // 创建结果数组
    const result = new Array(arr.length);
    
    // 从后往前遍历，保证稳定性
    for (let i = arr.length - 1; i >= 0; i--) {
        const index = arr[i] - min;
        result[count[index] - 1] = arr[i];
        count[index]--;
    }
    
    return result;
}
```

### 2. 桶排序
```javascript
// 桶排序 - 时间复杂度 O(n + k)
function bucketSort(arr, bucketSize = 5) {
    if (arr.length === 0) return arr;
    
    // 找到最大值和最小值
    const max = Math.max(...arr);
    const min = Math.min(...arr);
    
    // 计算桶的数量
    const bucketCount = Math.floor((max - min) / bucketSize) + 1;
    const buckets = new Array(bucketCount);
    
    // 初始化桶
    for (let i = 0; i < bucketCount; i++) {
        buckets[i] = [];
    }
    
    // 将元素分配到桶中
    for (let i = 0; i < arr.length; i++) {
        const bucketIndex = Math.floor((arr[i] - min) / bucketSize);
        buckets[bucketIndex].push(arr[i]);
    }
    
    // 对每个桶进行排序
    const result = [];
    for (let i = 0; i < bucketCount; i++) {
        if (buckets[i].length > 0) {
            // 使用插入排序对桶内元素排序
            insertionSort(buckets[i]);
            result.push(...buckets[i]);
        }
    }
    
    return result;
}
```

- **稳定排序**：冒泡排序、插入排序、归并排序、计数排序、桶排序
